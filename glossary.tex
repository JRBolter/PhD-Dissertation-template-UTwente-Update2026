\makeglossaries

\newglossaryentry{shotnoise}{
name=shot noise,
description={fundamental noise arising from Poisson statistics of photons},
}
\newglossaryentry{coma}{
	name=coma,
	description={aberration caused by off-axis light going through a lens},  
}
\newglossaryentry{bright field}{
	name=bright field,
	description={microscopy technique using white, collimated light to illuminate a big area of the sample and image the transmission of the sample.},
}
\newglossaryentry{rca}{
	name=RCA1 cleaning,
	description={cleaning using a basic solution of Ammonia, Hydroperoxide and water (1:1:5). Cleans most organic residues and leaves the surface negatively charged.},
}
\newglossaryentry{piranha}{
	name= PIRANHA cleaning,
	description={cleaning using a acidic solution of sulfuric acid and Hydroperoxide (4:1). Cleans all organic residues and leaves the surface positively charged.},
}
